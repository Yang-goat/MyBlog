            else
                xy = 1 - delta2;
                val = 2*(1 - u) + 2*(u - 0.5)*xy^(eta_m + 1);
                deltaq = 1 - val^mut_pow;
            end
            x_new(i) = x(i) + deltaq * (xu - xl);
            x_new(i) = min(max(x_new(i), xl), xu);
        end
    end
end
