# ========== æµ‹è¯•å‡½æ•° (ä»¥ ZDT1 ä¸ºä¾‹) ==========
def evaluate(pop):
    # pop: æ•°ç»„ shape=(N, dim)
    f1 = pop[:, 0]
    g = 1 + 9 * np.mean(pop[:, 1:], axis=1)
    f2 = g * (1 - np.sqrt(f1 / g))
    return np.vstack((f1, f2)).T  # shape = (N, 2)

def dominates(a, b):
    """åˆ¤æ–­ a æ˜¯å¦æ”¯é… bï¼ˆç›®æ ‡çš†æœ€å°åŒ–ï¼‰"""
    return np.all(a <= b) and np.any(a < b)

# ========== SPEA2 ä¸»å‡½æ•° ==========
def spea2(dim=30,
          pop_size=100,
          archive_size=50,
          max_gen=200,
          crossover_prob=0.9,
          mutation_prob=0.1,
          k=1):
    # åˆå§‹åŒ–ä¸»ç§ç¾¤
    pop = np.random.rand(pop_size, dim)
    archive = np.empty((0, dim))
    
    for gen in range(max_gen):
        # åˆå¹¶ä¸»ç§ç¾¤ä¸æ¡£æ¡ˆ
        union = np.vstack((pop, archive))
        fitness = evaluate(union)
        N = len(union)
        
        # å¼ºåº¦ S(i)
        S = np.zeros(N, dtype=int)
        for i in range(N):
            for j in range(N):
                if dominates(fitness[i], fitness[j]):
                    S[i] += 1
        
        # åŸå§‹é€‚åº”åº¦ R(i)
        R = np.zeros(N, dtype=int)
        for i in range(N):
            for j in range(N):
                if dominates(fitness[j], fitness[i]):
                    R[i] += S[j]
        
        # è·ç¦»çŸ©é˜µä¸å¯†åº¦ä¼°è®¡ D(i)
        dist = np.linalg.norm(fitness[:, None, :] - fitness[None, :, :], axis=2)
        np.fill_diagonal(dist, np.inf)  # è‡ªå·±åˆ°è‡ªå·±çš„è·ç¦»è®¾ä¸ºæ— ç©·
        sorted_dist = np.sort(dist, axis=1)
        sigma_k = sorted_dist[:, k]
        D = 1.0 / (sigma_k + 2.0)
        
        # ç»¼åˆé€‚åº”åº¦ F(i)
        F = R + D
        
        # ========== ç¯å¢ƒé€‰æ‹©ï¼šæ›´æ–°æ¡£æ¡ˆ ==========
        # å°† union æŒ‰ F ä»å°åˆ°å¤§æ’åºï¼Œå…ˆé€‰æ‹©éæ”¯é…è§£ï¼Œå†æŒ‰ F è¡¥è¶³ï¼æˆªæ–­è‡³ archive_size
        # é¦–å…ˆæå–éæ”¯é…é›†
        nondominated_idx = []
        for i in range(N):
            dominated = False
            for j in range(N):
                if dominates(fitness[j], fitness[i]):
                    dominated = True
                    break
            if not dominated:
                nondominated_idx.append(i)
        
        # æ„å»ºæ–°æ¡£æ¡ˆå€™é€‰
        if len(nondominated_idx) < archive_size:
            # è¡¥è¶³
            # å…ˆé€‰æ‰€æœ‰éæ”¯é…è§£
            sel = nondominated_idx.copy()
            # å†æŒ‰ F å‡åºé€‰å–å…¶ä½™
            remaining_idx = list(set(range(N)) - set(sel))
            remaining_idx.sort(key=lambda i: F[i])
            sel.extend(remaining_idx[:(archive_size - len(sel))])
        elif len(nondominated_idx) == archive_size:
            sel = nondominated_idx.copy()
        else:
            # è¶…è¿‡ï¼šéœ€è¦æˆªæ–­ â€” åˆ é™¤â€œæ‹¥æŒ¤â€ä¸ªä½“
            sel = nondominated_idx.copy()
            # è®¡ç®—å­é›† fitness_selã€distance å­çŸ©é˜µ
            while len(sel) > archive_size:
                # é‡å»ºè·ç¦»å­çŸ©é˜µ
                sub_fit = fitness[sel]
                sub_N = len(sel)
                sub_dist = np.linalg.norm(sub_fit[:, None, :] - sub_fit[None, :, :], axis=2)
                np.fill_diagonal(sub_dist, np.inf)
                # æ‰¾æœ€å°è·ç¦»å¯¹ï¼ˆi,jï¼‰
                minval = np.min(sub_dist)
                # æ‰¾å¯¹åº”ç´¢å¼•
                i_min, j_min = np.unravel_index(np.argmin(sub_dist), sub_dist.shape)
                # å“ªä¸ªä¸ªä½“æ›´â€œæ‹¥æŒ¤â€åˆ é™¤ï¼Ÿé€šå¸¸é€‰è·ç¦»å‘é‡æ›´å°è€…
                # ç®€åŒ–ï¼šåˆ é™¤ i_min
                del sel[i_min]
            # sel æœ€ç»ˆä¸º archive_size ä¸ªä½“
        archive = union[sel]
        fitness_archive = fitness[sel]
        
        # ========== äº¤é…é€‰æ‹© ==========
        # é‡‡ç”¨äºŒå…ƒé”¦æ ‡èµ›ï¼Œå€™é€‰æ± ä¸º archive
        parents = []
        for _ in range(pop_size):
            i1, i2 = random.sample(range(len(archive)), 2)
            if F[sel[i1]] < F[sel[i2]]:
                parents.append(archive[i1].copy())
            else:
                parents.append(archive[i2].copy())
        parents = np.array(parents)
        
        # ========== äº¤å‰ä¸å˜å¼‚ï¼ˆç®€åŒ–å½¢å¼ï¼‰ ==========
        # å¯¹æ¯å¯¹çˆ¶ä»£æ‰§è¡Œäº¤å‰ã€å˜å¼‚ç”Ÿæˆå­ä»£ pop
        new_pop = []
        for i in range(0, pop_size, 2):
            p1 = parents[i]
            p2 = parents[(i+1) % pop_size]
            # äº¤å‰
            if random.random() < crossover_prob:
                # ç®€åŒ–ï¼šå‡åŒ€äº¤å‰
                alpha = random.random()
                c1 = alpha * p1 + (1-alpha) * p2
                c2 = alpha * p2 + (1-alpha) * p1
            else:
                c1, c2 = p1.copy(), p2.copy()
            # å˜å¼‚
            for c in [c1, c2]:
                if random.random() < mutation_prob:
                    idx = random.randrange(dim)
                    c[idx] = random.random()
                new_pop.append(c)
        # ä¿è¯æ–°ç§ç¾¤å¤§å°ä¸º pop_size
        pop = np.array(new_pop[:pop_size])
        
        # å¯é€‰ï¼šæ‰“å°è¿›åº¦
        if gen % 50 == 0:
            print(f"Generation {gen}, Archive size = {len(archive)}")
    
    # è¾“å‡ºç»“æœ
    print("Final Pareto-approx archive size:", len(archive))
    return fitness_archive, archive

function SPEA2_main()
    % ============================================================
    %   Strength Pareto Evolutionary Algorithm 2 (SPEA2)
    %   ç»å…¸å®ç°ç¤ºä¾‹ â€”â€” ZDT1 æµ‹è¯•é—®é¢˜
    %   ä½œè€…ï¼šZitzler, Laumanns, Thiele (2001)
    % ============================================================

    % ========== å‚æ•°è®¾ç½® ==========
    N = 100;          % ä¸»ç§ç¾¤è§„æ¨¡
    N_A = 50;         % ç²¾è‹±æ¡£æ¡ˆè§„æ¨¡
    MaxGen = 200;     % æœ€å¤§è¿­ä»£æ¬¡æ•°
    Dim = 30;         % å†³ç­–å˜é‡ç»´æ•°
    k = 1;            % è¿‘é‚»å‚æ•°ï¼ˆç”¨äºå¯†åº¦ä¼°è®¡ï¼‰
    Pc = 0.9;         % äº¤å‰æ¦‚ç‡
    Pm = 0.1;         % å˜å¼‚æ¦‚ç‡

    % ========== åˆå§‹åŒ– ==========
    P = rand(N, Dim); % ä¸»ç§ç¾¤
    A = [];           % æ¡£æ¡ˆä¸ºç©º

    % ========== è¿­ä»£è¿›åŒ– ==========
    for gen = 1:MaxGen
        % åˆå¹¶ä¸»ç§ç¾¤ä¸æ¡£æ¡ˆ
        U = [P; A];
        Obj = evaluate(U); % è®¡ç®—ç›®æ ‡å‡½æ•°
        NumU = size(U,1);

        % ========== (1) è®¡ç®—å¼ºåº¦å€¼ S(i) ==========
        S = zeros(NumU,1);
        for i = 1:NumU
            for j = 1:NumU
                if dominates(Obj(i,:), Obj(j,:))
                    S(i) = S(i) + 1;
                end
            end
        end

        % ========== (2) è®¡ç®—åŸå§‹é€‚åº”åº¦ R(i) ==========
        R = zeros(NumU,1);
        for i = 1:NumU
            for j = 1:NumU
                if dominates(Obj(j,:), Obj(i,:))
                    R(i) = R(i) + S(j);
                end
            end
        end

        % ========== (3) è®¡ç®—å¯†åº¦é¡¹ D(i) ==========
        dist = pdist2(Obj, Obj);
        dist(1:NumU+1:end) = inf;    % å¯¹è§’çº¿è®¾ä¸ºæ— ç©·
        sortedDist = sort(dist, 2);
        sigma_k = sortedDist(:, k);
        Dval = 1 ./ (sigma_k + 2);

        % ========== (4) ç»¼åˆé€‚åº”åº¦ ==========
        Fit = R + Dval;

        % ========== (5) ç¯å¢ƒé€‰æ‹©ï¼ˆæ›´æ–°æ¡£æ¡ˆï¼‰ ==========
        % æå–éæ”¯é…è§£
        NonDom = [];
        for i = 1:NumU
            dominatedFlag = false;
            for j = 1:NumU
                if dominates(Obj(j,:), Obj(i,:))
                    dominatedFlag = true; break;
                end
            end
            if ~dominatedFlag
                NonDom = [NonDom; i];
            end
        end

        if length(NonDom) < N_A
            % è¡¥è¶³
            [~, idx] = sort(Fit);
            A_idx = unique([NonDom; idx(1:N_A-length(NonDom))]);
        elseif length(NonDom) == N_A
            A_idx = NonDom;
        else
            % æˆªæ–­æ“ä½œï¼šå¯†åº¦åˆ é™¤
            A_idx = NonDom;
            while length(A_idx) > N_A
                subObj = Obj(A_idx, :);
                subDist = pdist2(subObj, subObj);
                subDist(1:length(subDist)+1:end) = inf;
                [~, ind] = min(subDist(:));
                [row, ~] = ind2sub(size(subDist), ind);
                A_idx(row) = []; % åˆ é™¤æœ€æ‹¥æŒ¤ä¸ªä½“
            end
        end

        % æ›´æ–°æ¡£æ¡ˆ
        A = U(A_idx, :);
        ObjA = Obj(A_idx, :);

        % ========== (6) äº¤é…é€‰æ‹© ==========
        % äºŒå…ƒé”¦æ ‡èµ›é€‰æ‹©
        Parents = zeros(N, Dim);
        for i = 1:N
            i1 = randi(size(A,1));
            i2 = randi(size(A,1));
            if Fit(A_idx(i1)) < Fit(A_idx(i2))
                Parents(i,:) = A(i1,:);
            else
                Parents(i,:) = A(i2,:);
            end
        end

        % ========== (7) äº¤å‰ä¸å˜å¼‚ ==========
        Offspring = zeros(N, Dim);
        for i = 1:2:N
            p1 = Parents(i,:);
            p2 = Parents(mod(i,N)+1,:);
            % æ¨¡æ‹ŸäºŒè¿›åˆ¶äº¤å‰ (SBX)
            if rand < Pc
                beta = rand;
                c1 = 0.5*((1+beta)*p1 + (1-beta)*p2);
                c2 = 0.5*((1-beta)*p1 + (1+beta)*p2);
            else
                c1 = p1; c2 = p2;
            end
            % å¤šé¡¹å¼å˜å¼‚
            for c = [1 2]
                if rand < Pm
                    idx = randi(Dim);
                    if c==1, c1(idx)=rand; else, c2(idx)=rand; end
                end
            end
            Offspring(i,:) = c1;
            if i+1<=N, Offspring(i+1,:) = c2; end
        end

        % ========== (8) æ–°ç§ç¾¤ ==========
        P = Offspring;

        % å¯è§†åŒ–å½“å‰æ¡£æ¡ˆï¼ˆå¯é€‰ï¼‰
        if mod(gen, 50)==0
            scatter(ObjA(:,1), ObjA(:,2), 25, 'filled');
            title(['Generation ', num2str(gen)]);
            xlabel('f1'); ylabel('f2');
            drawnow;
        end
    end

    % ========== è¾“å‡ºç»“æœ ==========
    figure;
    scatter(ObjA(:,1), ObjA(:,2), 40, 'filled');
    title('Final Pareto Front (ZDT1, SPEA2)');
    xlabel('f_1'); ylabel('f_2');
end

    f2 = g .* (1 - sqrt(f1 ./ g));
    f = [f1, f2];
end
